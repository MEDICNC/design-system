getImageBrightness 함수의 return 값으로 0 ~ 250 사이의 숫자값 받게됨. <br/>
값이 높을수록 bright, 낮을수록 dark. <br/>
높고 낮음의 기준치 120

```jsx
<img  id="target"  src='img/image.png'/>

<script>
    const getImageBrightness = (base64, callback) => {
      const img = new Image();
      img.src = base64;

      img.style.display = "none";
      document.body.appendChild(img);

      var colorSum = 0;

      img.onload = function () {
        // create canvas
        var canvas = document.createElement("canvas");
        canvas.width = this.width;
        canvas.height = this.height;

        var ctx = canvas.getContext("2d");
        ctx.drawImage(this, 0, 0);

        var imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
        var data = imageData.data;
        var r, g, b, avg;

        for (var x = 0, len = data.length; x < len; x += 4) {
          r = data[x];
          g = data[x + 1];
          b = data[x + 2];

          avg = Math.floor((r + g + b) / 3);
          colorSum += avg;
        }

        var brightness = Math.floor(colorSum / (this.width * this.height));
        callback(brightness);
      };
    };

    const toDataURL = (url, callback) => {
      var xhr = new XMLHttpRequest();
      xhr.onload = function () {
        var reader = new FileReader();
        reader.onloadend = function () {
          callback(reader.result);
        };
        reader.readAsDataURL(xhr.response);
      };
      xhr.open("GET", url);
      xhr.responseType = "blob";
      xhr.send();
    };

    toDataURL( target.src, function(dataUrl) {
      getImageBrightness(dataUrl, (result)=> console.log(result))
    })
</script>
```
